```python
class CheckpointManager:
    """Format-agnostic checkpoint management for scientific workflows."""
    
    def __init__(self, checkpoint_dir: str = '.checkpoints'):
        """Initialize checkpoint manager with directory."""
        
    def save_checkpoint(self, stage: str, state: Dict[str, Any], 
                        metadata: Optional[Dict[str, Any]] = None) -> Path:
        """Save checkpoint with stage name, return path to checkpoint file."""
        
    def load_checkpoint(self, checkpoint_path: str = None, stage: str = None) -> Optional[Dict[str, Any]]:
        """Load checkpoint by path or stage name. Priority: path > stage > latest."""
        
    def list_checkpoints(self) -> List[Dict[str, Any]]:
        """List all available checkpoints with metadata."""
        
    def get_latest_checkpoint(self) -> Optional[Dict[str, Any]]:
        """Get the most recent checkpoint."""
        
    def get_checkpoint_by_stage(self, stage: str) -> Optional[Dict[str, Any]]:
        """Get checkpoint for specific stage."""
        
    def delete_checkpoint(self, checkpoint_path: str) -> bool:
        """Delete a checkpoint file."""
```

Implementation requirements:
1. Use JSON format for checkpoint files
2. Atomic writes: use tempfile.NamedTemporaryFile + os.rename (prevent corruption on crash)
3. Checkpoint filename format: `{stage}_checkpoint_{timestamp}.json`
4. Timestamp format: ISO 8601 (YYYYMMDD-HHMMSS)
5. Each checkpoint must contain: stage, timestamp, state, metadata, schema_version
6. Create checkpoint directory if it doesn't exist
7. CLI interface with subcommands:
   - `python -m scripts.infra.checkpoint save --stage NAME --state '{"key":"value"}'`
   - `python -m scripts.infra.checkpoint load --stage NAME`
   - `python -m scripts.infra.checkpoint list`
   - `python -m scripts.infra.checkpoint latest`
8. Support loading by explicit path: `--path /path/to/checkpoint.json`
9. schema_version = "1.0" for initial implementation

DO NOT:
- Use pickle (JSON only for human readability)
- Implement checkpoint expiration/cleanup (user manages)
- Auto-save (explicit save calls only)
- Validate checkpoint content on load (trust integrity)
