```python
from enum import Enum

class JobStatus(Enum):
    """Job status enumeration."""
    RUNNING = 'running'
    COMPLETED = 'completed'
    FAILED = 'failed'
    WARNING = 'warning'
    UNKNOWN = 'unknown'

class LogMonitor:
    """Monitor job execution via log parsing."""
    
    # Default error patterns for GROMACS/gnina/gmx_MMPBSA
    ERROR_PATTERNS = [...]
    WARNING_PATTERNS = [...]
    COMPLETION_MARKERS = [...]
    
    def __init__(self, log_file: str):
        """Initialize monitor with log file path."""
        
    def check_errors(self) -> List[str]:
        """Scan log file for error patterns, return list of error lines."""
        
    def check_warnings(self) -> List[str]:
        """Scan log file for warning patterns, return list of warning lines."""
        
    def check_completion(self) -> bool:
        """Check if job completed successfully."""
        
    def get_status(self) -> JobStatus:
        """Determine overall job status from log analysis."""
        
    def get_summary(self) -> Dict[str, Any]:
        """Get comprehensive log analysis summary."""
        
    def tail(self, n: int = 50) -> str:
        """Get last n lines of log file."""
        
    def grep(self, pattern: str) -> List[str]:
        """Search for pattern in log file."""
```

Implementation requirements:

1. **Error patterns (case-insensitive regex):**
   ```python
   ERROR_PATTERNS = [
       r'ERROR',
       r'Fatal error',
       r'Segmentation fault',
       r'SIGSEGV',
       r'Aborted',
       r'CUDA ERROR',
       r'GROMACS error',
       r'GPU memory allocation failed',
       r'Input validation failed',
       r'Assertion.*failed',
   ]
   ```

2. **Warning patterns:**
   ```python
   WARNING_PATTERNS = [
       r'WARNING',
       r'WARN',
       r'Note:',
       r'Step.*out of range',
   ]
   ```

3. **Completion markers (tool-specific):**
   ```python
   COMPLETION_MARKERS = [
       r'GROMACS.*finished',
       r'gnina.*finished',
       r'Performance:',  # GROMACS MD completion
       r'Final.*energy',
       r'Simulation complete',
       r'MMPBSA.*complete',
       r'Writing output',
   ]
   ```

4. **get_status() logic:**
   - If errors found → FAILED
   - Else if completion markers found → COMPLETED
   - Else if warnings found → WARNING
   - Else if file exists → RUNNING
   - Else → UNKNOWN

5. **get_summary() returns:**
   ```python
   {
       'status': str,
       'log_file': str,
       'errors': List[str],
       'warnings': List[str],
       'completed': bool,
       'error_count': int,
       'warning_count': int,
   }
   ```

6. **CLI interface:**
   - `python -m scripts.infra.monitor status --log /path/to/file.log`
   - `python -m scripts.infra.monitor errors --log /path/to/file.log`
   - `python -m scripts.infra.monitor warnings --log /path/to/file.log`
   - `python -m scripts.infra.monitor summary --log /path/to/file.log`
   - `python -m scripts.infra.monitor tail --log /path/to/file.log --lines 50`
   - `python -m scripts.infra.monitor grep --log /path/to/file.log --pattern "energy"`

7. Handle large log files efficiently (don't load entire file into memory for tail/grep)

DO NOT:
- Assume log files exist (handle missing files)
- Stop on first error (collect all errors)
- Modify log files
- Use external tools like grep/tail (pure Python implementation)
