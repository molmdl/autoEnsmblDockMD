class ConfigManager:
    """INI configuration file manager for pipeline settings."""
    
    def __init__(self, config_path: str):
        """Load configuration from file."""
        
    def get(self, section: str, key: str, fallback: Any = None) -> str:
        """Get string value with fallback."""
        
    def getint(self, section: str, key: str, fallback: int = None) -> int:
        """Get integer value with fallback."""
        
    def getfloat(self, section: str, key: str, fallback: float = None) -> float:
        """Get float value with fallback."""
        
    def getboolean(self, section: str, key: str, fallback: bool = None) -> bool:
        """Get boolean value with fallback."""
        
    def get_section(self, section: str) -> dict:
        """Get all key-value pairs from a section."""
        
    def get_execution_backend(self) -> str:
        """Get execution backend with auto-detection support."""
```

Implementation requirements:
1. Use Python's built-in `configparser` module (no external dependencies)
2. Support DEFAULT section inheritance
3. Handle missing sections/keys gracefully with fallbacks
4. Support 'auto' value for execution.backend that triggers environment detection
5. Include CLI interface: `python -m scripts.infra.config --config path/to/config.ini --section NAME --key KEY`
6. Return None (not exception) for missing values when no fallback provided
7. Type conversion errors should return fallback value, not raise

DO NOT:
- Add YAML support (INI only per SCRIPT-09 requirement)
- Implement interpolation (keep simple)
- Create a default config file location (config path must be explicit)

class AgentState:
    """File-based state management for agent session continuity."""
    
    def __init__(self, state_file: str = '.agent_state.json'):
        """Initialize state manager, load existing state if file exists."""
        
    def set(self, key: str, value: Any) -> None:
        """Set a state value and persist immediately."""
        
    def get(self, key: str, default: Any = None) -> Any:
        """Get a state value."""
        
    def update(self, updates: Dict[str, Any]) -> None:
        """Update multiple state values atomically."""
        
    def clear(self) -> None:
        """Clear all state."""
        
    def dump_to_file(self, output_path: str) -> None:
        """Dump state to specified file for session handoff."""
        
    def load_from_file(self, input_path: str) -> bool:
        """Load state from specified file."""
        
    def get_all(self) -> Dict[str, Any]:
        """Get entire state dictionary."""
```

Implementation requirements:
1. Use Python's built-in `json` module
2. Atomic writes: write to temp file, then rename (prevent corruption)
3. Include 'last_updated' timestamp automatically on every save
4. Load state from file on initialization if file exists
5. CLI interface: `python -m scripts.infra.state --file .agent_state.json --set key=value --get key`
6. Handle JSON serialization errors gracefully
7. Support nested dict updates via dot notation (e.g., 'stage.status' → state['stage']['status'])

DO NOT:
- Use pickle (JSON only for human readability)
- Implement locking (single-agent assumption for now)
- Auto-create state file on init (only on first set)
