import fcntl  # Add to imports at top of file

def save_checkpoint(self, stage: str, state: Dict[str, Any], 
                    metadata: Optional[Dict[str, Any]] = None) -> Path:
    """Save checkpoint with stage name, return path to checkpoint file.
    
    Uses exclusive file locking (fcntl.LOCK_EX) to prevent concurrent writes
    and ensure atomicity across multiple processes/agents.
    
    Args:
        stage: Name of the workflow stage (used in filename).
        state: State dictionary to persist (must be JSON-serializable).
        metadata: Optional metadata dictionary.
    
    Returns:
        Path to the created checkpoint file.
    
    Raises:
        OSError: If checkpoint directory cannot be created or file cannot be written.
        TypeError: If state or metadata is not JSON-serializable.
    """
    # Ensure directory exists
    self.checkpoint_dir.mkdir(parents=True, exist_ok=True)
    
    # Generate timestamp and filename
    timestamp = datetime.utcnow().strftime('%Y%m%d-%H%M%S')
    filename = f"{stage}_checkpoint_{timestamp}.json"
    checkpoint_path = self.checkpoint_dir / filename
    
    # Build checkpoint data
    checkpoint_data = {
        'stage': stage,
        'timestamp': datetime.utcnow().isoformat() + 'Z',
        'state': state,
        'metadata': metadata or {},
        'schema_version': self.SCHEMA_VERSION
    }
    
    # Atomic write with file locking
    with tempfile.NamedTemporaryFile(
        mode='w',
        dir=self.checkpoint_dir,
        delete=False,
        suffix='.json'
    ) as tmp:
        # Acquire exclusive lock on temp file
        fcntl.flock(tmp.fileno(), fcntl.LOCK_EX)
        
        json.dump(checkpoint_data, tmp, indent=2, default=str)
        tmp.flush()
        os.fsync(tmp.fileno())  # Force write to disk
        
        tmp_path = tmp.name
        
        # Lock is released when context manager exits
    
    # Atomic rename (replaces existing file if present)
    os.replace(tmp_path, checkpoint_path)
    
    return checkpoint_path
```

def _load_from_file(self, path: Path) -> Optional[Dict[str, Any]]:
    """Load checkpoint data from file with shared locking.
    
    Uses shared lock (fcntl.LOCK_SH) to allow concurrent reads but block
    during writes, preventing partial read corruption.
    
    Args:
        path: Path to checkpoint file.
    
    Returns:
        Full checkpoint dictionary or None if file is invalid/unreadable.
    """
    try:
        with open(path, 'r') as f:
            # Acquire shared lock (allows concurrent reads, blocks on writes)
            fcntl.flock(f.fileno(), fcntl.LOCK_SH)
            
            data = json.load(f)
            
            # Lock automatically released when context exits
        
        return data
    except (json.JSONDecodeError, IOError, OSError) as e:
        # Return None for invalid/missing files
        # Includes timeout if lock acquisition fails
        return None
```

This ensures:

- Reads block during writes (LOCK_EX from save_checkpoint)
- No partial JSON reads during checkpoint updates
