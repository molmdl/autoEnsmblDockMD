```python
def compute_contact_frequency(..., max_frames: int = 1000):
    """Compute contact frequency with optional frame striding.
    
    Args:
        max_frames: Maximum frames to analyze (default 1000). Stride
                    is auto-calculated to sample uniformly across trajectory.
                    Set to None to analyze all frames (slow for large trajectories).
    """
    contact_matrix = np.zeros((len(receptor), len(ligand)))

    # Calculate stride to sample ~max_frames uniformly
```

```python
def _plot_line(...):
    """Plot line graph with guaranteed figure cleanup.
    
    Uses try-finally to prevent memory leaks on exception.
    """
    fig = None
    try:
        fig, ax = plt.subplots(figsize=(8, 4))
        ax.plot(x, y, linewidth=1.2)
        ax.set_title(title)
        ax.set_xlabel(xlabel)
        ax.set_ylabel(ylabel)
        fig.tight_layout()
        fig.savefig(output, dpi=dpi)
    finally:
        if fig is not None:
            plt.close(fig)
```

```python
def _validate_coordinate_units(universe, expected_range=(0.1, 100.0)):
    """Warn if coordinates are outside expected range for nanometers.
    
    MDAnalysis universes should have positions in nanometers.
    This check detects likely unit mismatches (Angstroms, micrometers).
    
    Args:
        universe: MDAnalysis Universe object
        expected_range: (min, max) coordinate magnitude in nm
    """
    try:
        first_frame = universe.trajectory[0]
        coords = universe.atoms.positions
        
        coord_values = np.abs(coords.flatten())
        coord_min = np.min(coord_values[coord_values > 0])  # Exclude zeros
        coord_max = np.max(coord_values)
        
        if coord_min < expected_range[0] or coord_max > expected_range[1]:
            logging.warning(
                f"Coordinate range [{coord_min:.2f}, {coord_max:.2f}] nm is outside "
                f"expected range {expected_range}. Possible unit mismatch detected. "
                f"Verify your input files use nanometers."
            )
    except Exception as e:
        logging.debug(f"Could not validate coordinate units: {e}")
```
