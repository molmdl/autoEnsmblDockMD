```python
def _plot_line(...):
```

```python
def _validate_coordinate_units(universe, expected_range=(0.1, 100.0)):
    """Warn if coordinates are outside expected range for nanometers.
    
    MDAnalysis universes should have positions in nanometers.
    This check detects likely unit mismatches (Angstroms, micrometers).
    
    Args:
        universe: MDAnalysis Universe object
        expected_range: (min, max) coordinate magnitude in nm
    """
    try:
        first_frame = universe.trajectory[0]
        coords = universe.atoms.positions
        
        coord_values = np.abs(coords.flatten())
        coord_min = np.min(coord_values[coord_values > 0])  # Exclude zeros
        coord_max = np.max(coord_values)
        
        if coord_min < expected_range[0] or coord_max > expected_range[1]:
            logging.warning(
                f"Coordinate range [{coord_min:.2f}, {coord_max:.2f}] nm is outside "
                f"expected range {expected_range}. Possible unit mismatch detected. "
                f"Verify your input files use nanometers."
            )
    except Exception as e:
        logging.debug(f"Could not validate coordinate units: {e}")
```

```python
def run_command(self, command: List[str], cwd: str = None,
                env: Dict[str, str] = None, timeout: int = None) -> Tuple[int, str, str]:
    """Run command locally and capture output via file streaming.
    
    Streams stdout/stderr to temporary files instead of buffering in memory
    to prevent OOM on verbose long-running commands (GROMACS mdrun, MM/PBSA).
    """
    import tempfile
    
    # Merge environment variables
    run_env = os.environ.copy()
    if env:
        run_env.update(env)
    
    # Create temporary files for output streaming
    with tempfile.NamedTemporaryFile(mode='w+', delete=False, suffix='.stdout') as stdout_file, \
         tempfile.NamedTemporaryFile(mode='w+', delete=False, suffix='.stderr') as stderr_file:
        
        stdout_path = stdout_file.name
        stderr_path = stderr_file.name
        
        try:
            result = subprocess.run(
                command,
                cwd=cwd,
                env=run_env,
                stdout=stdout_file,  # Stream to file instead of capture_output=True
                stderr=stderr_file,
                text=True,
                timeout=timeout
            )
            
            # Read output from files (much smaller than full MD logs)
            stdout_file.seek(0)
            stderr_file.seek(0)
            
            stdout_content = stdout_file.read()
            stderr_content = stderr_file.read()
            
            return (result.returncode, stdout_content, stderr_content)
            
        except subprocess.TimeoutExpired as e:
            raise
        except FileNotFoundError as e:
            raise FileNotFoundError(f"Command not found: {command[0]}") from e
        finally:
            # Clean up temporary files
            try:
                Path(stdout_path).unlink(missing_ok=True)
                Path(stderr_path).unlink(missing_ok=True)
            except OSError:
                pass  # Best effort cleanup
```

```python
def _run_script(
    self,
    script: str,
    params: Dict[str, Any],
    env: Dict[str, str] = None,
) -> Dict[str, Any]:
    """Run a script command and capture execution details via file streaming."""
    import tempfile
    
    command_list = self._build_command(script, params)
    command_str = " ".join(command_list)

    merged_env = os.environ.copy()
    if env:
        merged_env.update(env)

    with tempfile.NamedTemporaryFile(mode='w+', delete=False, suffix='.stdout') as stdout_file, \
         tempfile.NamedTemporaryFile(mode='w+', delete=False, suffix='.stderr') as stderr_file:
        
        stdout_path = stdout_file.name
        stderr_path = stderr_file.name
        
        try:
            started = time.time()
            proc = subprocess.run(
                command_list,
                cwd=self.workspace,
                stdout=stdout_file,  # Stream to file
                stderr=stderr_file,
                text=True,
                env=merged_env,
            )
            duration = time.time() - started
            
            # Read output from files
            stdout_file.seek(0)
            stderr_file.seek(0)
            
            stdout_content = stdout_file.read()
            stderr_content = stderr_file.read()
            
            return {
                "returncode": proc.returncode,
                "stdout": stdout_content,
                "stderr": stderr_content,
                "command": command_str,
                "duration_seconds": duration,
            }
        finally:
            # Clean up temporary files
            try:
                Path(stdout_path).unlink(missing_ok=True)
                Path(stderr_path).unlink(missing_ok=True)
            except OSError:
                pass
```
