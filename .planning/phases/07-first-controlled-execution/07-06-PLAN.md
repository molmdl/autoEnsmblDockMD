**Step 1: Check for equilibration checkpoint**
```bash
cd work/test

if [[ -f .continue-here.md ]]; then
    echo "Checkpoint found - checking equilibration status..."
    
    job_ids=$(grep -A1 "^job_ids:" .continue-here.md | tail -1 | jq -r '.[]' 2>/dev/null | tr '\n' ',')
    job_ids=${job_ids%,}
    
    if [[ -n "$job_ids" ]]; then
        echo "Checking equilibration jobs: $job_ids"
        
        job_status=$(sacct -j "$job_ids" --format=State --noheader 2>/dev/null | sort -u)
        
        if echo "$job_status" | grep -q "RUNNING\|PENDING"; then
            echo "Equilibration still running. Cannot start production MD."
            exit 0
        elif echo "$job_status" | grep -q "FAILED\|TIMEOUT\|CANCELLED"; then
            echo "ERROR: Equilibration failed"
            exit 1
        elif echo "$job_status" | grep -q "COMPLETED"; then
            echo "✓ Equilibration completed"
        fi
    fi
fi
```

**Step 2: Verify complex system files**
```bash
for ligand in dzp ibp; do
    if [[ -d com/$ligand ]]; then
        echo "Checking $ligand complex system..."
        
        # Essential files for MD
        essential_files=(
            "com/$ligand/com.gro"
            "com/$ligand/sys.top"
            "com/$ligand/index.ndx"
        )
        
        missing=0
        for f in "${essential_files[@]}"; do
            if [[ -f "$f" ]]; then
                echo "  ✓ $f"
            else
                echo "  ✗ $f MISSING"
                missing=$((missing + 1))
            fi
        done
        
        if [[ $missing -gt 0 ]]; then
            echo "ERROR: $ligand complex system incomplete"
            exit 1
        fi
        
        # Check for .tpr (processed topology)
        if ls com/$ligand/*.tpr 2>/dev/null; then
            echo "  ✓ TPR file exists"
        else
            echo "  ⚠ No TPR file (will be generated by MD stage)"
        fi
    fi
done
```
