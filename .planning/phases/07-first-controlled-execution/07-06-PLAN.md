**Prerequisite Check:** Verify production MD completed before MM/PBSA.

**Step 1: Check MD checkpoint**
```bash
cd work/test

if [[ -f .continue-here.md ]]; then
    echo "Checkpoint found - checking MD job status..."
    
    job_ids=$(grep -A1 "^job_ids:" .continue-here.md | tail -1 | jq -r '.[]' 2>/dev/null | tr '\n' ',')
    job_ids=${job_ids%,}
    
    if [[ -n "$job_ids" ]]; then
        echo "Checking MD jobs: $job_ids"
        
        job_status=$(sacct -j "$job_ids" --format=State --noheader 2>/dev/null | sort -u)
        
        if echo "$job_status" | grep -q "RUNNING\|PENDING"; then
            echo "ERROR: Production MD still running"
            echo "MM/PBSA requires completed trajectories"
            echo "Wait for MD completion (check with sacct)"
            exit 0
        elif echo "$job_status" | grep -q "FAILED\|TIMEOUT\|CANCELLED"; then
            echo "ERROR: Production MD failed"
            sacct -j "$job_ids" --format=JobID,State,ExitCode
            exit 1
        elif echo "$job_status" | grep -q "COMPLETED"; then
            echo "✓ Production MD completed"
        fi
    fi
fi
```

**Step 2: Verify MD trajectories exist**
```bash
for ligand in dzp ibp; do
    if [[ -d com/$ligand ]]; then
        echo "Checking $ligand MD outputs..."
        
        # Check for trajectory
        if ls com/$ligand/*.xtc 2>/dev/null; then
            xtcs=$(ls com/$ligand/*.xtc 2>/dev/null | wc -l)
            echo "  ✓ Found $xtcs trajectory file(s)"
            
            # Check trajectory size (should be significant for production MD)
            for xtc in com/$ligand/*.xtc; do
                size=$(stat -f%z "$xtc" 2>/dev/null || stat -c%s "$xtc" 2>/dev/null)
                size_mb=$((size / 1024 / 1024))
                echo "    $xtc: ${size_mb}MB"
            done
        else
            echo "  ✗ No trajectory found for $ligand"
        fi
        
        # Check for energy file
        if ls com/$ligand/*.edr 2>/dev/null; then
            echo "  ✓ Energy file exists"
        fi
        
        # Check for TPR
        if ls com/$ligand/*.tpr 2>/dev/null; then
            echo "  ✓ TPR file exists"
        fi
    fi
done
```
