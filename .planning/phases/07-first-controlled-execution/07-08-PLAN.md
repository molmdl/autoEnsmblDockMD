**Step 1: Check MM/PBSA checkpoint**
```bash
cd work/test

if [[ -f .continue-here.md ]]; then
    echo "Checkpoint found - checking MM/PBSA job status..."
    
    job_ids=$(grep -A1 "^job_ids:" .continue-here.md | tail -1 | jq -r '.[]' 2>/dev/null | tr '\n' ',')
    job_ids=${job_ids%,}
    
    if [[ -n "$job_ids" ]]; then
        echo "Checking MM/PBSA jobs: $job_ids"
        
        job_status=$(sacct -j "$job_ids" --format=State --noheader 2>/dev/null | sort -u)
        
        if echo "$job_status" | grep -q "RUNNING\|PENDING"; then
            echo "ERROR: MM/PBSA still running"
            echo "Analysis can run in parallel with MM/PBSA, but typically done after"
            echo "Proceed with caution or wait for MM/PBSA completion"
        elif echo "$job_status" | grep -q "FAILED\|TIMEOUT\|CANCELLED"; then
            echo "WARNING: MM/PBSA failed"
            echo "Proceeding with trajectory analysis anyway"
        elif echo "$job_status" | grep -q "COMPLETED"; then
            echo "✓ MM/PBSA completed"
        fi
    fi
fi
```
