**Prerequisite Check:** Verify docking completed before proceeding to complex setup.

**Step 1: Check if resuming from checkpoint**
```bash
cd work/test

if [[ -f .continue-here.md ]]; then
    echo "Checkpoint found - checking job status..."
    
    # Extract job IDs from checkpoint
    job_ids=$(grep -A1 "^job_ids:" .continue-here.md | tail -1 | jq -r '.[]' 2>/dev/null | tr '\n' ',')
    job_ids=${job_ids%,}
    
    if [[ -n "$job_ids" ]]; then
        echo "Checking jobs: $job_ids"
        
        # Check job status
        job_status=$(sacct -j "$job_ids" --format=State --noheader | sort -u)
        
        if echo "$job_status" | grep -q "RUNNING\|PENDING"; then
            echo "ERROR: Jobs still running. Wait for completion."
            echo "Check: sacct -j $job_ids"
            exit 0  # Not an error, just not ready
        elif echo "$job_status" | grep -q "FAILED\|TIMEOUT\|CANCELLED"; then
            echo "ERROR: Jobs failed"
            sacct -j "$job_ids" --format=JobID,State,ExitCode
            exit 1
        elif echo "$job_status" | grep -q "COMPLETED"; then
            echo "✓ Jobs completed successfully"
        fi
    fi
fi
```

**Step 2: Verify docking outputs exist**
```bash
# Check for docking results
for ligand in ibp; do
    echo "Checking $ligand docking outputs..."
    
    if [[ -f dock/$ligand/docked.sdf ]]; then

    else
        echo "  ✗ dock/$ligand/docked.sdf MISSING"
        echo "  Cannot proceed with complex setup for $ligand"
    fi
    
    if [[ -f dock/$ligand/docking_scores.txt ]]; then
        echo "  ✓ dock/$ligand/docking_scores.txt exists"
    fi
done

# Must have at least one ligand docked
if [[ ! -f dock/ibp/docked.sdf ]]; then

fi
```
