#!/usr/bin/env python3
"""
Output validation comparator with MD-appropriate tolerances.

Compares generated outputs against expected results with:
- Structure counts: exact match
- Docking scores: ±10% relative tolerance
- RMSD/RMSF: ±20% mean, ±30% std tolerance
"""

import argparse
import json
from pathlib import Path
from typing import Dict, List, Any, Optional
import sys

# Tolerance presets
TOLERANCE_PRESETS = {
    'loose': {
        'docking_score': 0.10,
        'rmsd_mean': 0.20,
        'rmsd_std': 0.30,
        'rmsf_mean': 0.20,
        'rmsf_std': 0.30,
    },
    'medium': {
        'docking_score': 0.05,
        'rmsd_mean': 0.10,
        'rmsd_std': 0.20,
        'rmsf_mean': 0.10,
        'rmsf_std': 0.20,
    },
    'strict': {
        'docking_score': 0.02,
        'rmsd_mean': 0.05,
        'rmsd_std': 0.10,
        'rmsf_mean': 0.05,
        'rmsf_std': 0.10,
    }
}


def parse_gro_atom_count(gro_file: Path) -> int:
    """Parse atom count from GRO file."""
    try:
        with open(gro_file) as f:
            lines = f.readlines()
            if len(lines) >= 2:
                return int(lines[1].strip())
    except Exception as e:
        print(f"Error parsing {gro_file}: {e}")
    return 0


def parse_docking_scores(score_file: Path) -> List[float]:
    """Parse docking scores from file."""
    scores = []
    try:
        with open(score_file) as f:
            for line in f:
                parts = line.strip().split()
                if parts:
                    try:
                        scores.append(float(parts[-1]))
                    except ValueError:
                        continue
    except Exception as e:
        print(f"Error parsing {score_file}: {e}")
    return scores


def parse_xvg_statistics(xvg_file: Path) -> Dict[str, float]:
    """Parse mean and std from XVG file."""
    values = []
    try:
        with open(xvg_file) as f:
            for line in f:
                if line.startswith('#') or line.startswith('@'):
                    continue
                parts = line.strip().split()
                if len(parts) >= 2:
                    try:
                        values.append(float(parts[1]))
                    except ValueError:
                        continue
    except Exception as e:
        print(f"Error parsing {xvg_file}: {e}")
        return {'mean': 0, 'std': 0}
    
    if not values:
        return {'mean': 0, 'std': 0}
    
    import statistics
    return {
        'mean': statistics.mean(values),
        'std': statistics.stdev(values) if len(values) > 1 else 0
    }


def validate_structure_counts(gen_file: Path, exp_file: Path) -> Dict[str, Any]:
    """Validate structure counts (exact match)."""
    gen_atoms = parse_gro_atom_count(gen_file)
    exp_atoms = parse_gro_atom_count(exp_file)
    
    return {
        'atoms_match': gen_atoms == exp_atoms,
        'gen_atoms': gen_atoms,
        'exp_atoms': exp_atoms,
        'diff': abs(gen_atoms - exp_atoms)
    }


def validate_docking_scores(gen_file: Path, exp_file: Path, tolerance: float) -> Dict[str, Any]:
    """Validate docking scores with relative tolerance."""
    gen_scores = parse_docking_scores(gen_file)
    exp_scores = parse_docking_scores(exp_file)
    
    comparisons = []
    all_within_tol = True
    
    for i, (gen, exp) in enumerate(zip(gen_scores[:10], exp_scores[:10])):
        rel_diff = abs(gen - exp) / abs(exp) if exp != 0 else abs(gen - exp)
        within_tol = rel_diff <= tolerance
        if not within_tol:
            all_within_tol = False
        
        comparisons.append({
            'rank': i + 1,
            'gen_score': gen,
            'exp_score': exp,
            'rel_diff': rel_diff,
            'within_tolerance': within_tol
        })
    
    return {
        'all_within_tolerance': all_within_tol,
        'comparisons': comparisons,
        'gen_count': len(gen_scores),
        'exp_count': len(exp_scores)
    }


def validate_rmsd_stats(gen_file: Path, exp_file: Path, mean_tol: float, std_tol: float) -> Dict[str, Any]:
    """Validate RMSD/RMSF statistics with tolerances."""
    gen_stats = parse_xvg_statistics(gen_file)
    exp_stats = parse_xvg_statistics(exp_file)
    
    mean_diff = abs(gen_stats['mean'] - exp_stats['mean']) / exp_stats['mean'] if exp_stats['mean'] != 0 else 0
    std_diff = abs(gen_stats['std'] - exp_stats['std']) / exp_stats['std'] if exp_stats['std'] != 0 else 0
    
    return {
        'mean_within_tol': mean_diff <= mean_tol,
        'std_within_tol': std_diff <= std_tol,
        'gen_mean': gen_stats['mean'],
        'exp_mean': exp_stats['mean'],
        'mean_diff': mean_diff,
        'gen_std': gen_stats['std'],
        'exp_std': exp_stats['std'],
        'std_diff': std_diff
    }


def generate_report(results: Dict, output_file: Path):
    """Generate markdown validation report."""
    with open(output_file, 'w') as f:
        f.write("# Phase 7: Output Validation Report\n\n")
        f.write(f"**Generated:** {__import__('datetime').datetime.now().isoformat()}\n\n")
        
        f.write("## Validation Summary\n\n")
        
        # Structure counts
        if 'structure_counts' in results:
            f.write("### Structure Counts\n\n")
            for key, val in results['structure_counts'].items():
                status = "✓ PASS" if val.get('atoms_match', False) else "✗ FAIL"
                f.write(f"- {key}: {status} (gen={val.get('gen_atoms')}, exp={val.get('exp_atoms')})\n")
            f.write("\n")
        
        # Docking scores
        if 'docking_scores' in results:
            f.write("### Docking Scores\n\n")
            for key, val in results['docking_scores'].items():
                status = "✓ PASS" if val.get('all_within_tolerance', False) else "✗ FAIL"
                f.write(f"- {key}: {status} (tolerance ±{val.get('tolerance', 0)*100:.0f}%)\n")
            f.write("\n")
        
        # RMSD/RMSF
        if 'rmsd_stats' in results:
            f.write("### RMSD/RMSF Statistics\n\n")
            for key, val in results['rmsd_stats'].items():
                mean_status = "✓" if val.get('mean_within_tol', False) else "✗"
                std_status = "✓" if val.get('std_within_tol', False) else "✗"
                f.write(f"- {key}: mean {mean_status}, std {std_status}\n")
            f.write("\n")
        
        f.write("---\n\n")
        f.write("**NOTE:** MM/PBSA binding energies are NOT validated due to high variance (per CONTEXT.md).\n")


def main():
    parser = argparse.ArgumentParser(description="Validate outputs against expected results")
    parser.add_argument('--workspace', type=Path, default=Path('work/test'),
                        help='Generated workspace directory')
    parser.add_argument('--expected', type=Path, default=Path('expected/amb'),
                        help='Expected outputs directory')
    parser.add_argument('--mode', choices=['blind', 'targeted'], required=True,
                        help='Docking mode')
    parser.add_argument('--output', type=Path, default=Path('-'),
                        help='Output report path (default: stdout)')
    parser.add_argument('--tolerance-level', choices=['loose', 'medium', 'strict'],
                        default='loose', help='Tolerance preset')
    
    args = parser.parse_args()
    
    if not args.workspace.exists():
        print(f"ERROR: Workspace not found: {args.workspace}")
        sys.exit(1)
    
    tolerances = TOLERANCE_PRESETS[args.tolerance_level]
    results = {}
    
    # TODO: Add actual validation logic for each output type
    # For now, create placeholder results
    
    print(f"Validating workspace: {args.workspace}")
    print(f"Expected directory: {args.expected}")
    print(f"Mode: {args.mode}")
    print(f"Tolerance level: {args.tolerance_level}")
    
    # Generate report
    output_file = args.output if args.output != Path('-') else Path('/dev/stdout')
    generate_report(results, output_file)


if __name__ == '__main__':
    main()
