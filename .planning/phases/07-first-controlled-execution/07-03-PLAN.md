```bash
status=$(jq -r '.status' .handoffs/receptor_prep.json)
echo "Handoff status: $status"

case "$status" in
    success)
        echo "✓ Receptor ensemble preparation started successfully (job submitted)"
        ;;
    needs_review)
        echo "⚠ Warnings detected:"
        jq -r '.warnings[]?' .handoffs/receptor_prep.json
        ;;
    failure|blocked)
        echo "✗ Receptor preparation failed with status: $status"
        jq -r '.errors[]?' .handoffs/receptor_prep.json
        exit 1
        ;;
    *)
        echo "ERROR: Unknown status: $status"
        exit 1
        ;;
esac
```
